    "# WC 2026 Knockout Predictor\n",
    "\n",
    "A lightweight strength-rating + Poisson goal-simulation model for predicting\n",
    "all 16 knockout matches (Round of 16 -> Final) of the 2026 FIFA World Cup.\n",
    "\n",
    "**Method**\n",
    "1. Assign each team an Elo-style strength rating (0-100), based on current tournament form, squad quality, and historical pedigree.\n",
    "2. Convert the rating gap between two teams into expected goals (lambda) for each side.\n",
    "3. Sample a scoreline from independent Poisson distributions for each team's expected goals.\n",
    "4. If simulated scores tie, nudge the higher-rated team's score up by one (no draws allowed in a knockout).\n",
    "5. Propagate winners through the fixed FIFA bracket: R32 (undecided fixtures only) -> R16 -> QF -> SF -> 3rd place -> Final.\n",
    "6. Assign goal scorers by sampling from each team's known key attacking players.\n",
    "\n",
    "**Caveat**: this is a fun heuristic model for a prediction leaderboard, not a\n",
    "serious forecasting system. Exact scorelines and scorers are inherently\n",
    "high-variance and no model reliably beats chance on those columns."
   ]

    "def poisson(lam):\n",
    "    L = pow(2.71828182846, -lam)\n",
    "    k, p = 0, 1\n",
    "    while True:\n",
    "        k += 1\n",
    "        p *= random.random()\n",
    "        if p <= L:\n",
    "            return k - 1\n",
    "\n",
    "def sim_score(a, b):\n",
    "    diff = ratings[a] - ratings[b]\n",
    "    lam_a = max(0.4, 1.35 + diff/45)\n",
    "    lam_b = max(0.4, 1.15 - diff/45)\n",
    "    ga = min(5, poisson(lam_a))\n",
    "    gb = min(5, poisson(lam_b))\n",
    "    if ga == gb:\n",
    "        if ratings[a] >= ratings[b]:\n",
    "            ga += 1\n",
    "        else:\n",
    "            gb += 1\n",
    "    return ga, gb\n",
    "\n",
    "def pick_scorers(team, goals):\n",
    "    pool = scorers.get(team, [(9,\"Fwd\"),(10,\"Mid\"),(7,\"Wing\")])\n",
    "    chosen = random.sample(pool, min(goals, len(pool)))\n",
    "    extra = goals - len(chosen)\n",
    "    nums = [str(n) for n,_ in chosen]\n",
    "    while extra > 0:\n",
    "        nums.append(str(pool[0][0]))\n",
    "        extra -= 1\n",
    "    return \";\".join(nums) if goals > 0 else \"\"\n",
    "\n",
    "def play(a, b):\n",
    "    ga, gb = sim_score(a, b)\n",
    "    winner = a if ga > gb else b\n",
    "    return ga, gb, winner\n"
   ]
